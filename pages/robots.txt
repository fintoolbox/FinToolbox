// pages/robots.txt.js
export default function handler(req, res) {
  res.setHeader("Content-Type", "text/plain");
  // Cache for 1 day
  res.setHeader("Cache-Control", "public, max-age=86400, s-maxage=86400");

  const host = "https://fintoolbox.com.au";
  res.send(`User-agent: *
Allow: /

Sitemap: ${host}/sitemap.xml
`);
}
